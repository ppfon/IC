291// Controlador PI
292void Pifunc(sPI *reg, float T_div2, float Kp, float Ki, float satup, float satdown)
293{
294 reg->erro = reg->Xref - reg->Xm;
295
296 reg->erropi = reg->erro - (1/Kp)*reg->dif;
297
298 reg->inte = reg->inte_ant + T_div2 * (reg->erropi + reg->erropi_ant);
299 reg->inte_ant = reg->inte;
300 reg->erropi_ant = reg->erropi;
301
302 reg->piout = (Kp*reg->erro + Ki*reg->inte);
303
304 reg->piout_sat = reg->piout;
305 if(reg->piout>satup) reg->piout_sat = satup;
306 if(reg->piout<satdown) reg->piout_sat= satdown;
307
308 reg->dif = reg->piout - reg->piout_sat;
309}

312void SOGI_func(sSOGI *sog, float Tsample)
313{
314 sog->x = 4*Tsample*pi * sog->K_damp * sog->freq_res;
315 sog->y = (Tsample*2*pi * sog->freq_res)*(Tsample*2*pi * sog->freq_res);

317 sog->b1 = sog->x + sog->y + 4;
318 sog->b2 = sog->x - sog->y - 4;
319 sog->b3 = 2*(4 - sog->y);

321 sog->b0 = sog->x * (1/sog->b1);
322 sog->a1 = sog->b3 * (1/sog->b1);
323 sog->a2 = sog->b2 * (1/sog->b1);

325 sog->W = 4*Tsample*pi * sog->freq_res;

327 sog->V_sogi = sog->b0 * sog->Vm - sog->b0 * sog->Vm2 + sog->a1 * sog->V_sogi1+ sog->a2 * sog->V_sogi2;

329 sog->V_sogi_q = sog->W * sog->b0 * sog->Vm1 + sog->V_sogi_q1 * sog->a1 + sog->V_sogi_q2 * sog->a2;

341// Rampa
342void Ramp(sRamp *rmp, float sample)
343{
344 if(rmp->uin != rmp->y) rmp->t1 = rmp->t1 + sample;
345 else rmp->t1 = 0;
346 rmp->rate = (rmp->uin - rmp->y_ant)/(rmp->t1 - rmp->t1_ant);
347 if(rmp->rate > rmp->rising) rmp->y = (rmp->t1 - rmp->t1_ant)*rmp->rising + rmp->y_ant;
348 else if(rmp->rate < rmp->falling) rmp->y = (rmp->t1 - rmp->t1_ant)*rmp->falling + rmp->y_ant;
349 else rmp->y = rmp->uin;
350
351 rmp->t1_ant = rmp->t1;
352 rmp->y_ant = rmp->y;

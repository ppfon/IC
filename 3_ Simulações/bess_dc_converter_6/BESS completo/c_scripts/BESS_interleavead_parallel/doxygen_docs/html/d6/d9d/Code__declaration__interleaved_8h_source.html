88static inline void Ramp(sRamp *rmp, float sample) {
89 if (rmp->uin != rmp->y) rmp->t1 = rmp->t1 + sample; else rmp->t1 = 0;
90 rmp->rate = (rmp->uin - rmp->y_ant) / (rmp->t1 - rmp->t1_ant);
91 if (rmp->rate > rmp->rising) rmp->y = (rmp->t1 - rmp->t1_ant) * rmp->rising + rmp->y_ant;
92 else if (rmp->rate < rmp->falling) rmp->y = (rmp->t1 - rmp->t1_ant) * rmp->falling + rmp->y_ant;
93 else rmp->y = rmp->uin;
94 rmp->t1_ant = rmp->t1;
95 rmp->y_ant = rmp->y;
96}

107static inline void Pifunc(sPI *reg, float T_div2, float Kp, float Ki, float satup, float satdown) {
108 reg->erro = reg->Xref - reg->Xm;
109 reg->erropi = reg->erro - (1.0f / Kp) * reg->dif;
110 reg->inte = reg->inte_ant + T_div2 * (reg->erropi + reg->erropi_ant);
111 reg->inte_ant = reg->inte;

116 if (reg->piout < satdown) reg->piout_sat = satdown;
117 reg->dif = reg->piout - reg->piout_sat;

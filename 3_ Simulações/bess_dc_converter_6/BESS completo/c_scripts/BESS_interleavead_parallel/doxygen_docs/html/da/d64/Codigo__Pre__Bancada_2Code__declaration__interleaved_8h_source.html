118// Rampa
119void Ramp(sRamp *rmp, float sample)
120{
121 if(rmp->uin != rmp->y) rmp->t1 = rmp->t1 + sample;
122 else rmp->t1 = 0;
123 rmp->rate = (rmp->uin - rmp->y_ant)/(rmp->t1 - rmp->t1_ant);
124 if(rmp->rate > rmp->rising) rmp->y = (rmp->t1 - rmp->t1_ant)*rmp->rising + rmp->y_ant;
125 else if(rmp->rate < rmp->falling) rmp->y = (rmp->t1 - rmp->t1_ant)*rmp->falling + rmp->y_ant;
126 else rmp->y = rmp->uin;
127
128 rmp->t1_ant = rmp->t1;
129 rmp->y_ant = rmp->y;
130}
